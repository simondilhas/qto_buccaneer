 49    def get_property_value(self, element, set_name: str, prop_name: str) -> Optional[Any]:
 50        """
 51        Retrieves the value of a property or quantity from a specified Pset or Qset.
 52        Supports both IfcPropertySet and IfcElementQuantity.
 53
 54        Args:
 55            element: The IFC element to extract the property from.
 56            set_name (str): The name of the property set or quantity set (e.g. "Pset_SpaceCommon", "Qto_SpaceBaseQuantities").
 57            prop_name (str): The name of the property or quantity (e.g. "IsExternal", "NetFloorArea").
 58
 59        Returns:
 60            The unwrapped property value if found, otherwise None.
 61        """
 62        if element is None or not hasattr(element, "IsDefinedBy"):
 63            return None
 64
 65        for definition in element.IsDefinedBy:
 66            if not hasattr(definition, "RelatingPropertyDefinition"):
 67                continue
 68
 69            prop_def = definition.RelatingPropertyDefinition
 70            if prop_def is None:
 71                continue
 72
 73            # Process property sets
 74            if prop_def.is_a("IfcPropertySet") and prop_def.Name == set_name:
 75                for prop in getattr(prop_def, "HasProperties", []):
 76                    if prop.Name == prop_name:
 77                        if hasattr(prop, "NominalValue"):
 78                            val = prop.NominalValue
 79                            if hasattr(val, "wrappedValue"):
 80                                return val.wrappedValue
 81                            return val
 82                        elif hasattr(prop, "Value"):  # For simple props
 83                            val = prop.Value
 84                            if hasattr(val, "wrappedValue"):
 85                                return val.wrappedValue
 86                            return val
 87
 88            # Process quantity sets
 89            elif prop_def.is_a("IfcElementQuantity") and prop_def.Name == set_name:
 90                for quantity in getattr(prop_def, "Quantities", []):
 91                    if quantity.Name == prop_name:
 92                        if quantity.is_a("IfcQuantityArea"):
 93                            val = quantity.AreaValue
 94                        elif quantity.is_a("IfcQuantityVolume"):
 95                            val = quantity.VolumeValue
 96                        elif quantity.is_a("IfcQuantityLength"):
 97                            val = quantity.LengthValue
 98                        elif quantity.is_a("IfcQuantityCount"):
 99                            val = quantity.CountValue
100                        elif quantity.is_a("IfcQuantityWeight"):
101                            val = quantity.WeightValue
102                        elif hasattr(quantity, "NominalValue"):
103                            val = quantity.NominalValue
104                        else:
105                            val = None
106
107                        if hasattr(val, "wrappedValue"):
108                            return val.wrappedValue
109                        return val
110
111        return None

181        self,
182        filters: Optional[dict] = None,
183        filter_logic: Literal["AND", "OR"] = "AND",
184        ifc_entity: str = None
185    ) -> List[Any]:
186        """Get elements from the IFC file that match the given filters.
187        
188        Args:
189            filters: Dictionary of property filters
190            filter_logic: How to combine filters ("AND" or "OR")
191            ifc_entity: IFC entity type to filter for (e.g., "IfcSpace", "IfcWall")
192        
193        Returns:
194            List of matching IFC elements
195        """
196        # Get all elements of the specified type
197        if ifc_entity:
198            elements = self.model.by_type(ifc_entity)
199        else:
200            elements = self.model.by_type("IfcProduct")
201        
202        # If no filters, return all elements
203        if not filters:
204            return elements
205        
206        # Filter elements
207        filtered_elements = []
208        for element in elements:
209            matches = []
210            
211            for key, value in filters.items():
212                # Handle direct attribute comparison
213                if not "." in key:
214                    element_value = getattr(element, key, None)
215                    matches.append(self._compare_values(element_value, value))
216                    continue
217                
218                # Handle property set values
219                pset_name, prop_name = key.split(".")
220                prop_value = None
221                
222                # Check IfcPropertySet and IfcElementQuantity
223                for rel in getattr(element, "IsDefinedBy", []):
224                    definition = getattr(rel, "RelatingPropertyDefinition", None)
225                    if not definition:
226                        continue
227                        
228                    if definition.is_a("IfcPropertySet") and definition.Name == pset_name:
229                        for prop in definition.HasProperties:
230                            if prop.Name == prop_name:
231                                prop_value = getattr(prop, "NominalValue", None)
232                                if prop_value:
233                                    prop_value = prop_value.wrappedValue
234                                break
235                
236                    # Check IfcElementQuantity
237                    elif definition.is_a("IfcElementQuantity") and definition.Name == pset_name:
238                        for quantity in definition.Quantities:
239                            if quantity.Name == prop_name:
240                                if quantity.is_a("IfcQuantityArea"):
241                                    prop_value = quantity.AreaValue
242                                elif quantity.is_a("IfcQuantityLength"):
243                                    prop_value = quantity.LengthValue
244                                elif quantity.is_a("IfcQuantityVolume"):
245                                    prop_value = quantity.VolumeValue
246                                break
247                
248                matches.append(self._compare_values(prop_value, value))
249            
250            # Add element if it matches according to filter logic
251            if filter_logic == "AND" and all(matches):
252                filtered_elements.append(element)
253            elif filter_logic == "OR" and any(matches):
254                filtered_elements.append(element)
255

258    def _compare_values(self, actual_value: Any, filter_value: Any) -> bool:
259        # Handle string operators like "<=0.15"
260        if isinstance(filter_value, str):
261            operators = [">=", "<=", "!=", "=", ">", "<"]
262            for op in operators:
263                if filter_value.startswith(op):
264                    try:
265                        value = float(filter_value[len(op):])
266                        actual_value = float(actual_value)
267                        return {
268                            ">": actual_value > value,
269                            "<": actual_value < value,
270                            "=": actual_value == value,
271                            "!=": actual_value != value,
272                            "<=": actual_value <= value,
273                            ">=": actual_value >= value
274                        }[op]
275                    except (TypeError, ValueError):
276                        return False
277        
278        # Handle list with operator and value
279        elif isinstance(filter_value, list) and len(filter_value) == 2:
280            operator, value = filter_value
281            try:
282                actual_value = float(actual_value)
283                value = float(value)
284                return {
285                    ">": actual_value > value,
286                    "<": actual_value < value,
287                    "=": actual_value == value,
288                    "!=": actual_value != value,
289                    "<=": actual_value <= value,
290                    ">=": actual_value >= value
291                }[operator]
292            except (TypeError, ValueError):
293                return False
294        # Handle regular list membership
295        elif isinstance(filter_value, list):
296            return actual_value in filter_value
297        
298        return actual_value == filter_value

342            }
343            
344            # Get property and quantity sets
345            psets = self.get_property_sets(element)
346            
347            # Add properties and quantities with combined names

349                for prop_name, value in properties.items():

355            
356            data.append(element_data)

280        elif isinstance(filter_value, list) and len(filter_value) == 2:
281            operator, value = filter_value
282            try:
283                actual_value = float(actual_value)
284                value = float(value)
285                return {
286                    ">": actual_value > value,
287                    "<": actual_value < value,
288                    "=": actual_value == value,
289                    "!=": actual_value != value,
290                    "<=": actual_value <= value,
291                    ">=": actual_value >= value
292                }[operator]
293            except (TypeError, ValueError):
294                return False
295        # Handle regular list membership
296        elif isinstance(filter_value, list):

299        return actual_value == filter_value

205            return elements

208        filtered_elements = []
